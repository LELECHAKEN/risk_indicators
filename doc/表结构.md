CREATE OR REPLACE FORCE VIEW "RISK"."CRDS_VW_EVALUATION_D_RATING" ("债项代码", "债项名称", "债券内部编码", "持仓重要性", "评级级别", "评级日期", "评级状态", "处置建议", "公司代码", "公司名称", "主体评级", "行业一级代码", "行业一级名称", "行业二级代码", "行业二级名称", "是否有担保", "是否次级债", "是否永续债", "数据日期") AS 
  with t5 as (
/*有担保的债券且债项评级高于主体评级*/
select ed.debt_code
  from risk.crds_evaluation_m_rating_info m,
       risk.crds_evaluation_d_rating_info ed,
       risk.crds_system_inout_rating_info i,
       risk.jy_bond_code                  bc
 where m.company_code = ed.company_code
   and m.rating_status = 'RV'
   and ed.rating_status = 'RV'
   and to_number(ed.finally_rating) < to_number(m.finally_rating)
   and ed.finally_rating = i.rating_number
   and ed.innercode = bc.innercode
   and to_number(to_char(bc.interestenddate, 'yyyymmdd')) >=
       to_number(to_char(sysdate - 1, 'yyyymmdd')) /*排除到期债权*/
   and ed.debt_code in (select distinct d.debt_code
                          from risk.crds_evaluation_assure_info a,
                               risk.crds_company_debt_info      d
                         where a.debt_code = d.union_code))


select
t.DEBT_CODE as 债项代码,
t.DEBT_NAME as 债项名称,
t.INNERCODE as 债券内部编码,
( select  label from risk.crds_sys_dict where DESCRIPTION like '%持仓%' and value=(select IMPORTANCE  from  "RISK"."CRDS_COMPANY_POSITIONS_DEBT" p
where p.t_date = (select max(t_date) from "RISK"."CRDS_COMPANY_POSITIONS_DEBT" )
and p.debt_code=t8.debt_code ) ) as 持仓重要性,
t6.BRA_RATING as 评级级别,
t.rating_date  as 评级日期,
case when t.rating_status = 'RH' then '评级历史'
when t.rating_status = 'RV' then '当前有效'
when t.rating_status = 'SA' then '暂存'
when t.rating_status = 'CO' then '审批中'
when t.rating_status = 'RT' then '退回'
end  as 评级状态,
t.deal_proposal as 处置建议,
t.COMPANY_CODE as 公司代码,
t2.COMPANY_NAME as 公司名称,
t7.BRA_RATING  as 主体评级,
case when t3.indus_level = 1 then t3.INDUS_CODE else t4.INDUS_CODE end as 行业一级代码,
case when t3.indus_level = 1 then t3.INDUS_NAME else t4.INDUS_NAME end as 行业一级名称,

case when t9.SOTCK_TYPE = '永续债' then '是' else '否' end  as 是否永续债,

t.inserttime as 数据日期


from risk.crds_evaluation_d_rating_info t

 on t.COMPANY_CODE = t1.company_code
 and t1.rating_status='RV'

/*二级行业分类*/
